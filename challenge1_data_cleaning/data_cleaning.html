airbnb_final <- airbnb %>%
  select(-all_of(useless_features)) %>%
  
  #2.1 Convert date columns to format YYYY-MM-DD and class Date
  mutate(across(c(host_since, first_review, last_review), 
                ~ as.Date(.x, format ='%d/%m/%Y'))) %>%
  
  #2.2 Convert t/f columns to booleans
  mutate(across(c(host_is_superhost, host_has_profile_pic, host_identity_verified, 
                  is_location_exact, has_availability, instant_bookable), 
                ~ case_when(.x == 't' ~ TRUE,
                            .x == 'f' ~ FALSE,
                            .default = NA))) %>%
  
  #2.3 Convert currency columns (price and cleaning fee) to numeric
  mutate(across(c(price, cleaning_fee), ~ gsub('^\\$', '', .x))) %>%
  mutate(across(c(price, cleaning_fee), ~ gsub(',', '', .x))) %>%
  mutate(across(c(price, cleaning_fee), ~ as.numeric(.x))) %>%
  
  #2.4 Convert % columns (host_response_rate) to numeric
  mutate(host_response_rate = gsub('[^0-9.]', '', host_response_rate)) %>%  
  mutate(host_response_rate = as.numeric(host_response_rate)) %>%  

  #2.5 Convert all string features to lowercase and remove accents
  mutate(across(where(is.character), ~ stri_trans_general(tolower(.x), "Latin-ASCII")))

Exercise 3.1 Create as many boolean features as relevant for host_verifications

# Ensure `host_verifications` is a character column
airbnb_final$host_verifications <- as.character(airbnb_final$host_verifications)

# Handle NA values by replacing them with an empty string
airbnb_final$host_verifications[is.na(airbnb_final$host_verifications)] <- ''

# Remove unwanted characters (square brackets and single quotes)
host_verifications_cleaned <- gsub(pattern = "\\[|\\]|'", replacement = '', x = airbnb_final$host_verifications)

# Remove spaces inside the verification names
host_verifications_cleaned <- gsub(pattern = " ", replacement = '', x = host_verifications_cleaned)

# Combine values and keep separated using commas
host_verifications_cleaned <- paste(host_verifications_cleaned, collapse = ',')

# Split the host_verifications column into separate verification methods
host_verifications_cleaned <- unlist(str_split(host_verifications_cleaned, pattern = ','))

# Get unique verification methods
host_verifications_cleaned <- unique(host_verifications_cleaned)
host_verifications_cleaned <- host_verifications_cleaned[host_verifications_cleaned != '']  

# Create boolean columns for each verification method
list_host_verifications <- map(.x = host_verifications_cleaned, 
                                  .f = ~ as.integer(str_detect(airbnb_final$host_verifications, fixed(.x))))

# Rename columns properly
names(list_host_verifications) <- paste0('has_verification_', host_verifications_cleaned)

# Bind new boolean columns with the original dataset
airbnb_final <- bind_cols(airbnb_final, list_host_verifications)
